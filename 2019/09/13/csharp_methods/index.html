class NumberManipulator
{
public int factorial(int num)
{
/* 局部变量定义 */
int result;

    if (num == 1)
    {
        return 1;
    }
    else
    {
        result = factorial(num - 1) * num;
        return result;
    }
}

static void Main(string[] args)
{
    NumberManipulator n = new NumberManipulator();

  1. 参数传递包含（值参数，引用参数，输出参数）
  2. 值传递:传递实参变量的一个副本(copy),所以方法体内的操作不会改变该变量
  3. 引用传递:将实参变量直接传递给方法，方法体内的操作将改变该变量
  4. ref:能修改:调用之前先声明()，swap(ref int a,ref int b)
  5. out:必修改:调用之前可以不赋值，bool TryParse(string,out int)

按值传递参数

在函数内改变了值，值也没有发生任何的变化
class NumberManipulator
{
public void swap(int x, int y)
{
int temp;

   temp = x; /* 保存 x 的值 */
   x = y;    /* 把 y 赋值给 x */
   y = temp; /* 把 temp 赋值给 y */
}

static void Main(string[] args)
{
   NumberManipulator n = new NumberManipulator();
   /* 局部变量定义 */
   int a = 100;
   int b = 200;

   Console.WriteLine("在交换之前，a 的值： {0}", a);
   Console.WriteLine("在交换之前，b 的值： {0}", b);

   /* 调用函数来交换值 */
   n.swap(a, b);

   Console.WriteLine("在交换之后，a 的值： {0}", a);
   Console.WriteLine("在交换之后，b 的值： {0}", b);

   Console.ReadLine();
   output:
   在交换之前，a 的值：100
   在交换之前，b 的值：200
   在交换之后，a 的值：100
   在交换之后，b 的值：200
}

}

按引用传递参数

引用参数是对变量的内存位置的引用，不会创建一个新的存储位置，而值参数会创建，引用参数表示与提供给方法的实际参数具有相同的内存位置
使用ref声明引用参数

       class NumberManipulator
  {
  public void swap(ref int x, ref int y)
  {
     int temp;

     temp = x; /* 保存 x 的值 */
     x = y;    /* 把 y 赋值给 x */
     y = temp; /* 把 temp 赋值给 y */
   }

  static void Main(string[] args)
  {
     NumberManipulator n = new NumberManipulator();
     /* 局部变量定义 */
     int a = 100;
     int b = 200;

     Console.WriteLine("在交换之前，a 的值： {0}", a);
     Console.WriteLine("在交换之前，b 的值： {0}", b);

     /* 调用函数来交换值 */
     n.swap(ref a, ref b);

     Console.WriteLine("在交换之后，a 的值： {0}", a);
     Console.WriteLine("在交换之后，b 的值： {0}", b);

     Console.ReadLine();
     output:
     在交换之前，a 的值：100
     在交换之前，b 的值：200
     在交换之后，a 的值：200
     在交换之后，b 的值：100

  }
}

按输出传递参数

return用来返回函数中一个值，以使用 输出参数 来从函数中返回两个值。输出参数会把方法输出的数据赋给自己，其他方面与引用参数相似。

    class NumberManipulator
 {
 public void getValue(out int x )
 {
    int temp = 5;
    x = temp;
 }

 static void Main(string[] args)
 {
    NumberManipulator n = new NumberManipulator();
    /* 局部变量定义 */
    int a = 100;

    Console.WriteLine("在方法调用之前，a 的值： {0}", a);

    /* 调用函数来获取值 */
    n.getValue(out a);

    Console.WriteLine("在方法调用之后，a 的值： {0}", a);
    Console.ReadLine();

  }
 }
}
output:
在方法调用之前，a 的值： 100
在方法调用之后，a 的值： 5

提供给输出参数的变量不需要赋值。当需要从一个参数没有指定初始值的方法中返回值时，输出参数特别有用
